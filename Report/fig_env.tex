% add figure with label, description and reference
%\MijnFig[H]{width=5cm}{bestandsnaam}{Het bijschrift bij deze figuur}{label}
\newcommand{\MijnFig}[5][ht]{           	% Het eerste argument is standaar `ht'.
	\begin{figure}[#1]                      % Beginnen van de figure omgeving
		\begin{center}                      % Beginnen van de center omgeving
			\includegraphics[#2,keepaspectratio]{#3}        % Het eigenlijk invoegen van de figuur (2: opties, 3: bestandsnaam)
			\caption[#4]{#4\label{#5}}      % Het bijschrift (argument 4) en het label (argument 3)
		\end{center}
	\end{figure}
}

		\end{center}

}
